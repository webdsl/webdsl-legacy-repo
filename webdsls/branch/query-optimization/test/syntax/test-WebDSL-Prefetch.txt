define page root() {
  for(subA : EntA) {
    prefetch-for subA {
                    subs if(.propB > 2)
                    {
                      propC default [showSubs(EntB)] if(.propD>4)
                    }
                    propE
                    propF
                    propG default [showSubs(EntA),showSubs(EntB)]
                    EntD.propH if(.propB > 2)
                    {
                      EntD.propI if(EntD.propJ==1)
                    }
    }
    table {
      row {
        column {
          output(subA)
        }
      }
    }
  }
}
